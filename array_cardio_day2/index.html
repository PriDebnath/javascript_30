//how  Some and Every works in JavaScript

    const people = [
      { name: 'Ava', year: 1988 },
      { name: 'smith', year: 1986 },
      { name: 'jhon', year: 1970 },
      { name: 'Ana', year: 2015 }
        ];
        
    
    
    
    // Array.prototype.some() 
    
    // is at least one person 19?
    
    let isAdult = people.some((person)=>{
      let currentYear = new Date().getFullYear()
      
      if(currentYear - person.year >= 19 ){
        return true
      }
    })
    
    console.log(isAdult)
   
   //  true 
   // because one of them fullfil the criteria
    
    
    
    
    
    
    
    // Array.prototype.every()
    
    // is everyone 19?
    
        let isEveryoneAdult = people.every((person)=>{
      let currentYear = new Date().getFullYear()
      
      if(currentYear - person.year >= 19 ){
        return true
      }
    })
    
    console.log(isEveryoneAdult)
    
    // false
    // because  not everyone fullfiled the criteria
    
   
    
    
    
    
    // Array.prototype.find()
    
    // Find is like filter, but instead of returning all elephant who fulfill the criteria it just returns just  one you are looking for
    
    // find Ana
    
    
    const Ana = people.find((person)=> {
      return person.name == 'Ana'
    });
    
    console.log(Ana);
    
    // Array.prototype.findIndex()
    
    // Find index of Ana
    
    let indexOfAna = people.findIndex((person)=>{
      return person.name == 'Ana'
    })
    
    console.log('index of ana is ' +indexOfAna)
    
    // delete ana with his index
    
    people.splice(indexOfAna,1)

    console.table(people)
